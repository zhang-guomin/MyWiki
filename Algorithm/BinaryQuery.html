二分查找

array 数组里面查找 target (array必须是有序)
int left = 0,right = array.len - 1;
while (left <= right)
{
    int mid = (right - left)/2;
    if (target == array[mid])
        return true;
    else if array[mid] < target
        left = mid + 1;

C++代码示例

bool BinaryQuery(std::array<int, 10>& m_array, int& target)
{
    int iRight = m_array.size() - 1, iLeft = 0, iMid = 0;
    while (iRight >= iLeft)
    {
        iMid = (iRight + iLeft) / 2;
        if (target == m_array.at(iMid))
            return true;
        if (target > m_array.at(iMid))
            iLeft = iMid + 1;
        else
            iRight = iMid - 1;
    }

    return false;
}

C语言代码示例

int BinaryQuery(int x, int v[], int n)
{
    int low,high,mid;
    low = 0;
    high = n - 1;
    while(low <= high)
    {
        mid = (low + high) / 2;
        if (x < v[mid])
            high = mid - 1;
        else if (x > v[mid])
            low = mid + 1;
        else
            return mid;
    }
    return -1;
}
